void s2d::FixPoint::operator-=(const FixPoint &fp)
{
#if FP_VALIDATE
    //sign check
    int tmp = data - fp.data;

    if( data>0 && fp.data<0 )   assert( tmp > 0 );
    if( data<0 && fp.data>0 )   assert( tmp < 0 );
#endif

    data -= fp.data;
}
