void s2d::Draw2D::SetAlphaChannels(Texture *ptex)
{
    if(ptex!=NULL){

        if(!(ptex->GetFormat()&TF_ALPHA)){
            m_pAlphaChannel = NULL;
            return; 
        }
        if(ptex->GetWidth()!=m_pTexture->GetWidth() || 
            ptex->GetHeight()!=m_pTexture->GetHeight())
        {
            m_pAlphaChannel = NULL;
            return; 
        }
    }
    m_pAlphaChannel = ptex;
}
