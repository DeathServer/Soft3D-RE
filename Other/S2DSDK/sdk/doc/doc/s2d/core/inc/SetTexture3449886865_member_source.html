void s2d::Draw2D::SetTexture(Texture *ptex)
{
    if(ptex->GetFormat() & TF_ALPHA){
        m_pTexture = NULL;
        return; 
    }
    m_pTexture = ptex;

    // set colorkey
    if( ptex ) {
        EnableColorKey( (bool) ptex->GetHasColorKey() );
        u32 ckey = ptex->GetColorKey();
        SetColorKey( *(Pixel*)&ckey );
    }
}
