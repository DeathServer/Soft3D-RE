void s2d::StateMachine::EnterState(State< TStateMachine > *pState)
    {
        // enter parent states first before enter current state
        State<TStateMachine>* pParentState = pState->GetParent();
        if( pParentState ) {
            EnterState( pParentState );
        }
        pState->Enter( (TStateMachine*)this );
    }
