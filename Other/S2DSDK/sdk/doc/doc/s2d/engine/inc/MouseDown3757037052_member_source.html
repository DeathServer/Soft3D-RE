void s2d::Input::MouseDown(MouseButton mb,s32 x, s32 y)
{
    if( !(m_MouseTbl[ mb ] & MS_DOWN) ) {
        m_MouseTbl[ mb ] = MS_DOWN;
        m_MouseTbl[ mb ] |= MS_FIRSTDOWN;
    } else {
        m_MouseTbl[ mb ] = MS_DOWN;
    }
    m_MouseX=x;
    m_MouseY=y;
    //m_bIsAnyKeyDown = true;
}
