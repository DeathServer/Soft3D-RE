void s2d::Input::KeyDown(Keys key)
{
    if( !(m_KeyTbl[ key ] & KS_DOWN) ) {
        m_KeyTbl[ key ] = KS_DOWN;
        m_KeyTbl[ key ] |= KS_FIRSTDOWN;
    } else {
        m_KeyTbl[ key ] = KS_DOWN;
        m_KeyPressTbl[ key ] = m_KeyPressTbl[ key ] + 1;
        if( m_KeyPressTbl[ key ] >= 2 ) {
            m_KeyTbl[ key ] |= KS_PRESS;
            m_KeyPressTbl[ key ] = 0;
        }
    }
    m_bIsAnyKeyDown = true;
}
