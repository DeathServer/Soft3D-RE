f32 tan(s32 ang)
{
#ifdef FP_VALIDATE
    assert( ang < S2D_ANGRANGE );
#endif

    if( ang & S2D_ANG_HALFPI ){
        return -TanLUT[S2D_ANG_HALFPI - (ang & S2D_ANG_QUADMASK)];    //tg(PI/2+a) = -ctg(a)      II
                                                                    //tg(3PI/2+a) = -ctg(a)     IV
    }else{
        return TanLUT[ang & S2D_ANG_QUADMASK];                        //tg(a)                     I
                                                                    //tg(PI+a) = tg(a)          III
    }
}
